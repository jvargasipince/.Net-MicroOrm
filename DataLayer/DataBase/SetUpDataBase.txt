use DemoOrm

create table dbo.invoice (
id int identity(1,1) primary key,
nroinvoice char(5) not null,
company nvarchar(100) not null,
customer nvarchar(100) not null,
ammount money not null,
nroproducts int not null,
datecreate datetime default getdate()
)
go

create table dbo.invoiceDetail (
id int identity(1,1) primary key,
idInvoice int foreign key references invoice(id),
productname nvarchar(50),
quantity int,
unitprice money,
subtotal money
)
go

--Create Stored Procedures

create procedure dbo.GetFullInvoice
@id int
as
BEGIN 

	Select 
		id,  
		nroinvoice, 
		company, 
		customer, 
		ammount, 
		nroproducts, 
		datecreate 
	from 
		dbo.invoice 
	where Id = @Id;

	Select 
		id, 
		productname, 
		quantity, 
		unitprice, 
		subtotal 
	from 
		dbo.invoiceDetail 
	where  
	idInvoice = @Id;

END
go

create procedure dbo.InsertInvoice
@id int output,
@nroinvoice char(5),
@company nvarchar(100),
@customer nvarchar(100),
@ammount money,
@nroproducts int
as 
BEGIN
	insert into invoice 
		(nroinvoice, company, customer, ammount, nroproducts)
	values 
		(@nroinvoice, @company, @customer, @ammount, @nroproducts);
		
	SET @id = CAST(scope_identity() as int)
	
END
go
